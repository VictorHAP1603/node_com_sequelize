findAll()  === retorna tudo
    -- parametros
    1°: findAll({
        attributes: Um array informando quais colunas devem trazer, ex ['name', 'age]
    })

    2°: findAll({
        attributes: {exclude: 'Remove aq a coluna que nao quer q venha'}
    })
    

    3°: findAll({
        where: Aqui tem varias condições e jeitos de se fazer
    })
